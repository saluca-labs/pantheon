function generateId(): string {
  return Math.random().toString(36).slice(2, 10);
}

function formatTime(ts: number): string {
  return new Date(ts * 1000).toLocaleTimeString([], {
    hour: "2-digit",
    minute: "2-digit",
  });
}

function formatDate(ts: number): string {
  return new Date(ts * 1000).toLocaleDateString([], {
    month: "short",
    day: "numeric",
  });
}

// ---------------------------------------------------------------------------
// SSE streaming helper
// ---------------------------------------------------------------------------

interface StreamCallbacks {
  onAction: (result: string) => void;
  onToken: (token: string) => void;
  onDone: (sessionId: string, confidence: number, escalated: boolean) => void;
  onError: (error: string) => void;
}

async function streamChat(
  message: string,
  history: { role: string; content: string }[],
  sessionId: string | undefined,
  callbacks: StreamCallbacks,
  signal: AbortSignal
): Promise<void> {
  const soulkey = getStoredSoulKey();
  const headers: Record<string, string> = {
    "Content-Type": "application/json",
    Accept: "text/event-stream",
  };
  if (soulkey) headers["Authorization"] = `Bearer ${soulkey}`;

  const resp = await fetch(`${config.apiBaseUrl}/v1/support/chat`, {
    method: "POST",
    headers,
    body: JSON.stringify({ message, history, session_id: sessionId }),
    signal,
  });

  if (!resp.ok || !resp.body) {
    callbacks.onError(`Request failed: ${resp.status}`);
    return;
  }

  const reader = resp.body.getReader();
  const decoder = new TextDecoder();
  let buffer = "";

  while (true) {
    const { done, value } = await reader.read();
    if (done) break;

    buffer += decoder.decode(value, { stream: true });
    const lines = buffer.split("\n");
    buffer = lines.pop() ?? "";

    let currentEvent = "";
    let currentData = "";

    for (const line of lines) {
      if (line.startsWith("event: ")) {
        currentEvent = line.slice(7).trim();
      } else if (line.startsWith("data: ")) {
        currentData = line.slice(6).trim();
      } else if (line === "" && currentEvent && currentData) {
        try {
          const parsed = JSON.parse(currentData);
          if (currentEvent === "action") {
            callbacks.onAction(parsed.result as string);
          } else if (currentEvent === "token") {
            callbacks.onToken(parsed.token as string);
          } else if (currentEvent === "done") {
            callbacks.onDone(
              parsed.session_id as string,
              parsed.confidence as number,
              parsed.escalated as boolean
            );
          } else if (currentEvent === "error") {
            callbacks.onError(parsed.error as string);
          }
        } catch {
          // malformed SSE data -- skip
        }
        currentEvent = "";
        currentData = "";
      }
    }
  }
}

function MessageBubble({ msg }: { msg: ChatMessage }) {
  const isUser = msg.role === "user";
  const isAction = msg.role === "action";

  if (isAction) {
    return (
      <div className="mx-3 my-1.5">
        <div className="bg-of-primary/8 border border-of-primary/20 rounded-lg px-3 py-2">
          <p className="text-[10px] font-semibold text-of-primary uppercase tracking-wider mb-1">
            Action Result
          </p>
          <pre className="text-[11px] text-of-on-surface-variant whitespace-pre-wrap font-mono leading-relaxed">
            {msg.content}
          </pre>
        </div>
      </div>
    );
  }

  return (
    <div className={`flex gap-2 px-3 py-1.5 ${isUser ? "flex-row-reverse" : "flex-row"}`}>
      {/* Avatar */}
      <div
        className={`w-6 h-6 rounded-full flex items-center justify-center shrink-0 mt-0.5 ${
          isUser
            ? "bg-of-primary/15 border border-of-primary/25"
            : "bg-of-surface-container-highest border border-of-outline-variant/20"
        }`}
      >
        {isUser ? (
          <User className="w-3 h-3 text-of-primary" />
        ) : (
          <Bot className="w-3 h-3 text-of-on-surface-variant" />
        )}
      </div>

      {/* Bubble */}

export default function ChatWidget() {
  const [open, setOpen] = useState(false);
  const [panel, setPanel] = useState<Panel>("chat");
  const [messages, setMessages] = useState<ChatMessage[]>([]);
  const [input, setInput] = useState("");
  const [streaming, setStreaming] = useState(false);
  const [sessionId, setSessionId] = useState<string | undefined>(undefined);
  const [escalated, setEscalated] = useState(false);
  const [sessions, setSessions] = useState<SessionSummary[]>([]);
  const [loadingHistory, setLoadingHistory] = useState(false);
  const [unreadCount, setUnreadCount] = useState(0);

  const bottomRef = useRef<HTMLDivElement>(null);
  const inputRef = useRef<HTMLTextAreaElement>(null);
  const abortRef = useRef<AbortController | null>(null);

  // Auto-scroll to bottom
  useEffect(() => {
    bottomRef.current?.scrollIntoView({ behavior: "smooth" });
  }, [messages]);

  // Focus input when panel opens
  useEffect(() => {
    if (open && panel === "chat") {
      setTimeout(() => inputRef.current?.focus(), 150);
    }
  }, [open, panel]);

  // Load history when switching to history panel
  useEffect(() => {
    if (panel === "history" && open) {
      setLoadingHistory(true);
      api
        .get("/v1/support/chat/history")
        .then((data: unknown) => {
          const d = data as { sessions: SessionSummary[] };
          setSessions(d.sessions ?? []);
        })
        .catch(() => setSessions([]))
        .finally(() => setLoadingHistory(false));
    }
  }, [panel, open]);

  // Reset unread on open
  useEffect(() => {
    if (open) setUnreadCount(0);
  }, [open]);

  const loadSession = useCallback(async (sid: string) => {
    try {
      const data = (await api.get(`/v1/support/chat/history/${sid}`)) as {
        session_id: string;
        turns: { role: string; content: string; timestamp: number }[];
      };
      const restored: ChatMessage[] = data.turns.map((t) => ({
        id: generateId(),
        role: t.role as "user" | "assistant",
        content: t.content,
        timestamp: t.timestamp,
      }));
      setMessages(restored);
      setSessionId(data.session_id);
      setPanel("chat");
    } catch {
      // silently ignore
    }
  }, []);

  const sendMessage = useCallback(async () => {
    const text = input.trim();
    if (!text || streaming) return;

    setInput("");
    setEscalated(false);

    const userMsg: ChatMessage = {
      id: generateId(),
      role: "user",
      content: text,
      timestamp: Date.now() / 1000,
    };

    const assistantId = generateId();
    const assistantMsg: ChatMessage = {
      id: assistantId,
      role: "assistant",
      content: "",
      timestamp: Date.now() / 1000,
      streaming: true,
    };

    setMessages((prev) => [...prev, userMsg, assistantMsg]);
    setStreaming(true);

    // Build history for context (last 10 messages, exclude current streaming)
    const historyForRequest = messages.slice(-10).map((m) => ({
      role: m.role === "action" ? "assistant" : m.role,
      content: m.content,
    }));

    const abort = new AbortController();
    abortRef.current = abort;

    try {
      await streamChat(
        text,
        historyForRequest,
        sessionId,
        {
          onAction: (result) => {
            const actionMsg: ChatMessage = {
              id: generateId(),
              role: "action",
              content: result,
              timestamp: Date.now() / 1000,
            };
            setMessages((prev) => {
              // Insert before the streaming assistant message
              const idx = prev.findIndex((m) => m.id === assistantId);
              if (idx === -1) return [...prev, actionMsg];
              const next = [...prev];
              next.splice(idx, 0, actionMsg);
              return next;
            });
          },
          onToken: (token) => {
            setMessages((prev) =>
              prev.map((m) =>
                m.id === assistantId
                  ? { ...m, content: m.content + token }
                  : m
              )
            );
          },
          onDone: (sid, _confidence, wasEscalated) => {
            setSessionId(sid);
            setEscalated(wasEscalated);
            setMessages((prev) =>
              prev.map((m) =>
                m.id === assistantId ? { ...m, streaming: false } : m
              )
            );
            setStreaming(false);
            if (!open) setUnreadCount((c) => c + 1);
          },
          onError: (err) => {
            setMessages((prev) =>
              prev.map((m) =>
                m.id === assistantId
                  ? { ...m, content: `Error: ${err}`, streaming: false }
                  : m
              )
            );
            setStreaming(false);
          },
        },
        abort.signal
      );
    } catch (err: unknown) {
      if ((err as { name?: string }).name !== "AbortError") {
        setMessages((prev) =>
          prev.map((m) =>
            m.id === assistantId
              ? { ...m, content: "Connection error. Please try again.", streaming: false }
              : m
          )
        );
      }
      setStreaming(false);
    }
  }, [input, streaming, messages, sessionId, open]);

      <AnimatePresence>

            transition={{ type: "spring", stiffness: 300, damping: 30 }}

          >
            {/* Header */}
            <div className="flex items-center justify-between px-4 py-3 border-b border-of-outline-variant/15 shrink-0">

                <div className="w-7 h-7 rounded-full bg-of-primary/15 border border-of-primary/25 flex items-center justify-center">

                  <p className="text-[9px] text-of-on-surface-variant/60 uppercase tracking-wider">
